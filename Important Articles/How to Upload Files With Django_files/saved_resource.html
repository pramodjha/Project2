                                                                                                                                                                  Kemal Oztay KILL It WithFire a year ago

                                                                                                                                                                  Yes I found a solution. The solution is to write my own storage. Here 's the code of my solution:

                                                                                                                                                                  from django.core.files.storage import FileSystemStorage
                                                                                                                                                                  import os

                                                                                                                                                                  class XMLFileSystemStorage(FileSystemStorage):
                                                                                                                                                                      def __init__(self, xmlpath):
                                                                                                                                                                          self.xmlpath = xmlpath
                                                                                                                                                                          super(XMLFileSystemStorage, self).__init__(location=os.path.join(settings.STATIC_ROOT, self.xmlpath), base_url=settings.STATIC_URL+self.xmlpath)

                                                                                                                                                                      def __eq__(self, other):
                                                                                                                                                                              return self.xmlpath == other.xmlpath

                                                                                                                                                                  xmlStorage = XMLFileSystemStorage('xml/upload')

                                                                                                                                                                  class ImporterFile(models.Model):
                                                                                                                                                                      """
                                                                                                                                                                      This model keeps the file information.
                                                                                                                                                                      """
                                                                                                                                                                      file = models.FileField(storage=xmlStorage) 
                                                                                                                                                                      remote_url = models.URLField(blank=True, null=True)
                                                                                                                                                                      uploaded_at = models.DateTimeField(auto_now_add=True)


                                                                                                                                                                  I hope it helps...
